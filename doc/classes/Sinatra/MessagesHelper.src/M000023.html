    def pretty(*args)
      
      puts ""
      
      if args[0].class == Hash
        return self.process_hash(args[0])
      end
      
      args.each do |arg|

        title = "DEBUG|#{arg.class}"
        if arg.respond_to?('inspect')
          value = arg.inspect
        else
          value = arg
        end

        puts self.format_message(value, title)
        self.add_to_console(self.format_for_console(value, title))
        
      end
    
      puts ""
      
    end
